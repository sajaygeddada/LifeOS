📋 Supabase SQL — run once in SQL Editor

CREATE TABLE IF NOT EXISTS public.vyrona_data (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users NOT NULL UNIQUE,
  user_email text,
  data jsonb DEFAULT '{}'::jsonb,
  updated_at timestamptz DEFAULT now()
);
ALTER TABLE public.vyrona_data ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users own data" ON public.vyrona_data
  FOR ALL USING (auth.uid() = user_id);
ALTER PUBLICATION supabase_realtime ADD TABLE public.vyrona_data;
